 28    def bit_to_random_code(self, bitchr: str) -> int:
 29        """
 30        Generates a random integer code between [0, 255] based on the given bitvalue.
 31        
 32        Args:
 33            bitchr (str): Bit character in string representation.
 34
 35        Returns:
 36            Random integer code according to the bitvalue.
 37            Bitvalue: 0 => returns random integer in [0, 127].
 38            Bitvalue: 1 => returns random integer in [128, 255].
 39        """
 40        if bitchr == "0":
 41            return random.randint(0, 127)
 42        else:
 43            return random.randint(128, 255)
 44
 45    def code_to_bit(self, code: int) -> str:
 46        """
 47        Returns the bitvalue corresponding to the given code.
 48        
 49        Args:
 50            code (int): Integer code between [0, 255]
 51
 52        Returns:
 53            Bit character in string representation according to the given code.
 54            Code: [0, 127] => returns "0".
 55            Code: [128, 255] => returns "1".
 56        """
 57        if 0 <= code <= 127:
 58            return "0"
 59        elif 128 <= code <= 255:
 60            return "1"

 62    def send(self, log_file_name, dst_ip, dst_port) -> None:
 63        """
 64        - A random binary message is constructed and logged into the file.
 65        - The bits of the binary message are processed, 2 bits in each pass.
 66        - Each bit is transformed into randomly generated 8-bit code.
 67        - The codes for the 2 bits are combined into a single 16-bit value using bitshifting.
 68        - A packet is constructed and this value is used as the TCP source port value which is a 16-bit field.
 69        - Therefore, 2 bits of the original message are transmitted per each packet.
 70        - The tested throughput using github codespaces is: 128 bits in 2.298 secs => 55.7 bps
 71
 72
 73        Each bit is encoded as follows:
 74        - A random integer code is generated based on the given bitvalue.
 75        - If the bitvalue is 0: a random integer in [0, 127].
 76        - If the bitvalue is 1: a random integer in [128, 255].
 77        """
 78        binary_message = self.generate_random_binary_message_with_logging(log_file_name)
 79
 80        # Traverse through the binary message with a stride of 2.
 81        for i in range(0, len(binary_message), 2):
 82            # Extract the 2 bits from the binary message starting at the offset i and encode them.
 83            bit1_code = self.bit_to_random_code(binary_message[i])
 84            bit2_code = self.bit_to_random_code(binary_message[i + 1])
 85
 86            # Combine the two encoded values (8-bits each) into single value 16 bit
 87            # We achieve this by shifting the first encoded value by 8 bits to left
 88            # and add the second encoded value.
 89            src_port = (bit1_code << 8) + bit2_code
 90
 91            # Make TCP packet, put the encoded value in TCP source port field and send.
 92            ip = IP(dst=dst_ip)
 93            tcp = TCP(sport=src_port, dport=dst_port)
 94            super().send(ip / tcp)
 95
 96    def recv_message_builder(self, packet) -> None:
 97        # Extract the encoded value from the TCP source port field.
 98        src_port = packet["TCP"].sport
 99
100        # Extract the first encoded value by shifting 8-bits to left.
101        bit1_code = src_port >> 8
102        # Extract the second encoded value by selecting on right-most 8 bits.
103        bit2_code = src_port & 0xFF
104
105        # Convert the value to the corresponding bitvalues.
106        bit1 = self.code_to_bit(bit1_code)
107        bit2 = self.code_to_bit(bit2_code)
108
109        # Append the bitvalues to bitstring
110        self.cur_bitstring += bit1
111        self.cur_bitstring += bit2
112
113        # Increment bit count
114        self.cur_bitcnt += 2
115
116        # If all the bits of character are received
117        if self.cur_bitcnt == 8:
118            char = self.convert_eight_bits_to_character(self.cur_bitstring)
119            # Signal the receiver to stop receiving after character received is a full-stop "."
120            if char == ".":
121                self.stop_flag = True
122            # Append the character to the message being built
123            self.message += char
124
125            # Reset the bitstring state for the new character to be read
126            self.cur_bitcnt = 0
127            self.cur_bitstring = ""
128
129    def receive(self, log_file_name, src_ip, dst_ip, dst_port) -> None:
130        """
131        - Sniff for the packets filtered by their transport type: TCP, source IP: src_ip, destination IP: dst_ip, destination port: dst_port
132        - Process each packet with `recv_message_builder` callback.
133        - Stop the packet if `self.stop_flag` is set to True
134
135        The processing of each packet is done as follows:
136        - The encoded value from the TCP source port field is extracted.
137        - Two encoded values are decoded corresponding to two bits of the message.
138        - The bitvalues are appended to the bitstring which keeps track of all the bits for the current character being received
139        - If the length of bitstring reaches 8, meaning the character is fully received. The bitstring is converted from binary to character and appended to the message being received. 
140        - The receiver stops after reading the full-stop "." as it would be the last character of the message.
141        """
142        pktfilter = (
143            "tcp and src host "
144            + src_ip
145            + " and dst host "
146            + dst_ip
147            + " and dst port "
148            + str(dst_port)
149        )
150        sniff(
151            filter=pktfilter,
152            prn=self.recv_message_builder,
153            stop_filter=lambda x: self.stop_flag,
154        )
155        # The message is logged after being fully received.
156        self.log_message(self.message, log_file_name)
